   SELECT u.*, ou.UnitName  
                FROM [User] u  
                LEFT OUTER JOIN [Unit] ou ON ou.UnitCode = u.UnitCode  
                WHERE UserId = @userId  
                SELECT rf.FunctionCode  
                FROM ezAcquire.dbo.[User] u  
                JOIN ezAcquire.dbo.UserRoleMap ru ON ru.UserId = u.UserId  
                JOIN ezAcquire.dbo.RoleFunctionMap rf ON rf.RoleCode = ru.RoleCode  
                WHERE u.UserId = @userId
