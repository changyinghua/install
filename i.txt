declare @formUid varchar(10) = '2005780230'
declare @version tinyint = 2

-- 元鎂 DB
SELECT

    fp.Form_UID
   ,fp.VERSION
   ,fp.DisplayOrder 
   ,p.FileName

FROM

    LinkFormPage fp
    JOIN Pages p ON p.Page_UID = fp.Page_UID

WHERE

    fp.Form_UID = @formUid
    AND fp.Version = @version

ORDER BY
   
    fp.DisplayOrder
